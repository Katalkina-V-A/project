def check_resident_type
  if self.resident_type == 'Student'
    "студент"
  elsif self.resident_type == "Workman"
    "сотрудник"
  elsif self.resident_type == "Guest"
    "гость"
  elsif self.resident_type == "Relative"
    "родственник"
  end
end
